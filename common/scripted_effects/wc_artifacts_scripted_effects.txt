	new_artifact = {
		if = {
			limit = { quality == 1 }
			set_description = "artifact_common_origin"
		}
		else_if = {
			limit = { quality == 2 }
			set_description = "artifact_uncommon_origin"
		}
		else_if = {
			limit = { quality == 3 }
			set_description = "artifact_rare_origin"
		}
		else_if = {
			limit = { quality == 4 }
			set_description = "artifact_epic_origin"
		}
		else = { set_description = "artifact_legendary_origin" }
		give_artifact_random_name_effect = yes
		set_creation_date = 1.1.1
		log = "[This.OriginalOwner.GetTitledNameWithNick] has had [This.GetName] added to their Treasury."
	}
