				AND = {
					prestige = 500

						NOT = { true_religion_group = ROOT }

				}

		}

					}

			}

				modifier = ruler_converted_from
			}
			if = {
				limit = {
					religion_group = christian

						NOT = { true_religion_group = ROOT }
					}

			}

						NOT = { true_religion_group = ROOT }
					}

					}
				}

					chance = 60

				}
			}
			if = {

					religion_group = zoroastrian_group
