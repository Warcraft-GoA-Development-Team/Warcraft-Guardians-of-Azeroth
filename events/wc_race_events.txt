		# Picks mother's religion and culture if child is born in prison
		if = {
			limit = {
				NOT = { age = 1 }
				prisoner = yes
				event_target:target_mother = { always = yes }
			}
			religion = event_target:target_mother
			culture = event_target:target_mother
