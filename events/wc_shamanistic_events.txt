# k_firelands declares war if you change religion
character_event = {
	id = WCSHM.8050
	
	is_triggered_only = yes
	hide_window = yes
	
	only_playable = yes
	only_independent = yes
	
	trigger = {
		NOT = { religion = fire_cult }
		FROM = { religion = fire_cult }
	}
	
	immediate = {
		capital_scope = {
			k_firelands = {
				holder_scope = {
					spawn_firelord_troops_effect = yes
					unsafe_war = {
						casus_belli = change_religion_cb
						target = ROOT
					}
				}
			}
		}
	}
}
