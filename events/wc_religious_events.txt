character_event = {
	id = WCREL.7000
	desc = EVTDESC_WCREL_7000
	picture = GFX_evt_shadow_acolytes
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	capable_only = yes
	prisoner = no
	religion = voidfarer
	
	is_triggered_only = yes
	
	trigger = {
		NOT = {
			personality_traits = 5
		}
		NOT = {
			trait = entropean
		}
		NOT = {
			trait = inbred
		}
		is_dumb_trigger = no
		NOT = {
			trait = infirm
		}
		NOT = {
			has_alternate_start_parameter = {
				key = religion_names
				value = random
			}
		}
	}
	
	weight_multiplier = {
		days = 1
		
		modifier = {
			factor = 1.7
			trait = scholar
		}
		modifier = {
			factor = 1.7
			trait = zealous
		}
		modifier = {
			factor = 0.5
			trait = cynical
		}
		modifier = {
			factor = 0.5
			trait = slothful
		}
		modifier = {
			factor = 0.5
			trait = content
		}
	}

	option = {
		name = EVTOPTA_WCREL_7000
		set_character_flag = aspiring_entropean_flag
		add_character_modifier = {
			name = aspiring_entropean
			duration = -1
		}
	}
	option = {
		name = EVTOPTB_WCREL_7000
		if = {
			limit = {
				NOT = {
					trait = slothful
				}
			}
			random = {
				chance = 20
				add_trait = slothful
				hidden_tooltip = {
					character_event = {
						id = 38251 #Notify Slothful
					}
				}
			}
		}
		if = {
			limit = {
				trait = slothful
			}
			piety = -10
		}
	}
}
